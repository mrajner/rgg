%% \documentclass[ 
%% startpage = 10,
%% ...,
%% ]{rgg}

\PassOptionsToClass{
  % select the number of starting page, volume, year and doi
  % if logodg argument is present than deGruyter logo is printed at first page
  startpage = 11,
  volume    = 123,
  year      = 2015,
  doi       = 152.rgg/abcd,
  review    = false,
  logodg,
}{rgg}

%% put proper path for article
\input{./rgg_sample_article.tex}
